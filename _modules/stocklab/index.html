[docs]def eval(di_str):
    """
    Evaluate the DataIdentifier string.

    :param di_str: the DataIdentifier.
    :type di_str: str
    """
    names = di_str.split('.')
    node_name = names[0]
    fields = [field.split(':') for field in names[1:]]
    fields = {k:v for k,v in fields}
    node = get_node(node_name)
    return node(**fields)
